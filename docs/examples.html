1. Basic numeric intervals

// Require the package from npm (CommonJS on RunKit) const { default: IntervalTree, Interval } = require("@flatten-js/interval-tree"); const tree = new IntervalTree(); [[6,8],[1,4],[5,12],[1,1],[5,7]].forEach(p => tree.insert(p)); console.log("Keys (sorted):", tree.keys); console.log("Search [2,3]:", tree.search([2,3]));

2. TimeInterval (Date endpoints)

const { default: IntervalTree, TimeInterval } = require("@flatten-js/interval-tree"); const day = (s) => new Date(s); const tree = new IntervalTree(); const a = new TimeInterval(day('2020-01-01T00:00:00Z'), day('2020-01-31T00:00:00Z')); const b = new TimeInterval(day('2020-01-15T00:00:00Z'), day('2020-02-15T00:00:00Z')); tree.insert(a, 'A'); tree.insert(b, 'B'); const hits = tree.search(new TimeInterval(day('2020-01-10T00:00:00Z'), day('2020-01-20T00:00:00Z'))); console.log(hits); // ['A','B']

3. Interval2D (lexicographic 2D)

const { default: IntervalTree, Interval2D } = require("@flatten-js/interval-tree"); const tree = new IntervalTree(); const r1 = new Interval2D([0,0], [10,10]); const r2 = new Interval2D([5,5], [15,15]); const r3 = new Interval2D([20,0], [25,5]); tree.insert(r1, 'R1'); tree.insert(r2, 'R2'); tree.insert(r3, 'R3'); console.log('Keys:', tree.keys); console.log('Search [7,7]-[8,8]:', tree.search(new Interval2D([7,7],[8,8])));
